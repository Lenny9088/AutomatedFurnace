alias activate d0
alias typeTemp d1
alias quantityPressure d2
alias typeDial d3
alias QuantityDial d4
alias furnace d5
alias type r0
alias quantity r1
alias isActive r2
alias reagents r3
alias expectedReagents r4
alias maxTemp r5
alias maxPressure r6
alias minTemp r7
alias minPressure r8
alias currentTemp r9
alias currentPressure r10
alias ticks r11
waitActive:
move ticks 0
move maxTemp 2000
move maxPressure 10000
move minTemp 1000
move minPressure 1000
s typeTemp Setting 0
s quantityPressure Setting 0
s furnace Lock 1
l isActive activate Setting
l type typeDial Setting
l quantity QuantityDial Setting
beq isActive 1 selectOres
yield
j waitActive
selectOres:
mul expectedReagents 100 quantity
beq type 0 waitActive
beq quantity 0 waitActive
s furnace Open 0
beq type 8 constantan
beq type 13 electrum
beq type 14 invar
beq type 15 solder
beq type 16 steel
standardIngot: 
mul expectedReagents 50 quantity
push type
push quantity
j startOre
steel:
mul expectedReagents 200 quantity
push 8
push quantity
mul quantity 3 quantity
push 3
push quantity
j startOre
constantan:
move maxTemp 1400
push 1
push quantity
push 5
push quantity
j startOre
electrum:
move maxPressure 2300
push 7
push quantity
push 3
push quantity
j startOre
invar:
move maxPressure 6900
move minPressure 6100
move minTemp 1250
push 3
push quantity
push 5
push quantity
j startOre
solder:
move maxPressure 3400
push 8
push quantity
push 3
push quantity
j startOre
startOre:
beq sp 0 heat
pop quantity
pop type
s typeTemp Setting type
s quantityPressure Setting quantity
waitOre:
l quantity quantityPressure Setting
l type typeTemp Setting 
beq quantity 0 startOre
beq type -1 fail
yield
j waitOre
heat:
s furnace Lock 0
s typeTemp Setting maxTemp
s quantityPressure Setting maxPressure
waitSmelt:
yield
l currentTemp furnace Temperature
l currentPressure furnace Pressure
blt currentTemp minTemp waitSmelt
blt currentPressure minPressure waitSmelt
bgt currentTemp maxTemp waitSmelt
bgt currentPressure maxPressure waitSmelt
l reagents furnace Reagents
beq reagents expectedReagents empty
bgt ticks expectedReagents empty
add ticks 3 ticks
j waitSmelt
fail:
yield
l currentTemp furnace Temperature
bgt currentTemp 300 fail
empty:
s furnace Lock 0
s furnace Open 1
sleep 10
l reagents furnace Reagents
beq reagents 0 waitActive
yield
j empty
